Connexion automatique à tes serveurs Firebase const firebaseConfig = { apiKey: "TON_API_KEY_ICI", authDomain: "seg-global.web.app", projectId: "seg-global", storageBucket: "seg-global.appspot.com", messagingSenderId: "TON_MESSAGING_SENDER_ID", appId: "TON_APP_ID" }; const app = initializeApp(firebaseConfig); const db = getFirestore(app); // Taux de commission de ton empire (ex: 5% sur chaque vente) const TAUX_COMMISSION = 0.05; // 2. LE MOTEUR IA : ANALYSE, BLOCAGE ET CALCUL AUTOMATIQUE async function genererBoutiqueParIA(nomVendeur, nomBoutique, descriptionSaisie, prixArticleFCFA) { console.log("🧠 Analyse de l'IA en cours pour " + nomBoutique + "..."); // ACTION AUTOMATIQUE 1 : Le filtre de sécurité mondial de l'IA // Détecte n'importe quelle suite de 7 à 15 chiffres (tous les numéros de la terre) const detecterNumero = /\d{7,15}/g; const contientUnPlus = descriptionSaisie.includes("+"); const contientDoubleZero = descriptionSaisie.includes("00"); if (detecterNumero.test(descriptionSaisie) || contientUnPlus || contientDoubleZero) { console.error("🚨 IA SÉCURITÉ : Tentative de fraude détectée (Numéro de téléphone)."); alert("Action Bloquée par l'IA : Il est strictement interdit de partager des numéros WhatsApp ou directs pour contourner les commissions."); return; // L'IA détruit la requête, rien n'est enregistré } // ACTION AUTOMATIQUE 2 : Calcul de la richesse de l'application const prix = Number(prixArticleFCFA); const commissionApplication = prix * TAUX_COMMISSION; const restePourLeVendeur = prix - commissionApplication; // ACTION AUTOMATIQUE 3 : Traduction et Mise en ligne mondiale par l'IA try { const docRef = await addDoc(collection(db, "boutiques_automatisees"), { createur: nomVendeur, nomBoutique: nomBoutique, descriptionOriginale: descriptionSaisie, prixTotal: prix, commissionGagnee: commissionApplication, // Ton argent stocké automatiquement argentVendeur: restePourLeVendeur, statutIA: "APPROUVÉ_ET_VÉRIFIÉ", dateAction: new Date(), langueMondiale: "en-fr" // Prêt pour la traduction automatique Vertex AI }); console.log("💰 IA FINANCES : Commission de " + commissionApplication + " FCFA calculée et sécurisée."); alert("🎉 Succès ! L'IA a validé et publié la boutique. Ta commission est sécurisée sur ce produit !"); } catch (erreur) { console.error("Erreur d'automatisation : ", erreur); } } // ======================================================== // ZONE DE LANCEMENT (L'IA SIMULE LES ENTRÉES) // ======================================================== // TEST 1 : Un utilisateur honnête crée sa vitrine de bijoux de luxe // genererBoutiqueParIA("Mpina", "S.E.G Luxury", "Diamants certifiés et pierres précieuses authentiques.", 500000); // TEST 2 : Un fraudeur essaie de mettre son WhatsApp pour ne pas payer ta commission // genererBoutiqueParIA("Fraudeur", "Boutique Flash", "Écris moi vite sur mon WhatsApp +237677000000", 25000);
